\section*{Problem 1}

% Problem 1i
\noindent i.  $f(n) = \sqrt{2^{7n}} \textrm{ , } g(n) = \lg (7^{2n})$

\vspace{3mm}

\indent $f(n) = \sqrt{2^{7n}} = (2^{7n})^{\frac{1}{2}} = 2^{\frac{7}{2}n} = (2^{\frac{7}{2}})^n = a^n \implies f(n)$ is exponential
\indent $g(n) = \lg (7^{2n}) = 2n\lg (7) = (2\lg 7)n = cn \implies g(n)$ is linear
\indent Since f(n) is exponential and g(n) is linear $\implies \boxed{f  = \Omega(g)}$

\vspace{3mm}

% Problem 1ii
\noindent ii. $f(n) = 2^{n\ln n} \textrm{ , } g(n) = n!$

\vspace{3mm}

\indent $\lg (f(n)) = \lg 2^{n\ln n} = n\ln n \implies \lg (f(n))$ is linearithmic
\indent $\lg (g(n)) = \lg (n!) \approx n \lg n \implies \lg (g(n))$ is linearithmic. Note that the approximation of $\lg (n!)$ is done using Stirling's approximation.
\indent Since $\lg (f(n))$ and $\lg (g(n))$ are both linearithmic $\implies \boxed{f(n) = \Theta(g(n))}$

\vspace{3mm}

% Problem 1iii
\noindent iii. $f(n) = \lg (\lg ^{*}n) \textrm{ , } g(n) = \lg ^*(\lg n)$

\vspace{3mm}

\indent $f(n) = \lg (\lg ^{*}n = \lg (1+ \lg (\lg (n))$. In the preceding function we can ignore the 1 because it is constant and does not grow with n, so we can say that $f(n) = O(\lg (\lg (\lg (n))))$
\indent $g(n) = \lg ^{*}(\lg (n) = 1 + \lg (\lg (\lg (n))$. In the preceding function we can ignore the 1 because it is constant and does not grow with n, so we can say that $g(n) = O(\lg (\lg (\lg (n))))$
\indent Since f(n) and g(n) grow at the same rate as a function of the input $\implies \boxed{f(n) = \Theta(g(n))}$

\vspace{3mm}

% Problem 1iv
\noindent iv. $f(n) = \frac{\lg n^2}{n} \textrm{ , } g(n) = \lg ^*n$

\vspace{3mm}

\indent $f(n) =  \frac{\lg n^2}{n} = \frac{2\lg (n)}{n}$. Since n dominates $\lg n$ the preceding function will approach 0 as the input approaches $\infty$. (Possibly the only program in the world that gets faster as the input gets larger)
\indent $g(n) = \lg ^{*}(n)$. g(n) is a very slow growing function, but still grows as a function of its input unlike g(n).
\indent Since f(n) takes almost no time to execute for large n and g(n) grows with n, g(n) will dominate f(n) $\implies \boxed{f(n) = O(g(n))}$

\vspace{3mm}

% Problem 1v
\noindent v. $f(n) = 2^n \textrm{ , } g(n) = n^{\lg n}$

\vspace{3mm}

\indent $f(n) = 2^n \implies \lg (f(n)) = \lg (2^n) = n\cdot \lg (2)$ is linear with respect to the input.
\indent $g(n) = n^{\lg n} \implies \lg (g(n)) = \lg (n^{\lg n}) = (\lg (n))^2$ is polylogarithmic with respect to the input.
\indent Since polynomial (linear is a polynomial) functions dominate polylogarithmic functions $\implies \boxed{f(n) = \Omega(g(n))}$ 

\vspace{3mm}

% Problem 1vi
\noindent vi. $f(n) = 2^{\sqrt{\lg n}} \textrm{ , } g(n) = n(\lg n)^3$

\vspace{3mm}

\indent $f(n) = 2^{\sqrt{\lg n}} \implies \lg (2^{\sqrt{\lg n}}) = {\sqrt{\lg n}}\cdot \lg 2$ is polylogarithmic with respect to input
\indent $g(n) = n(\lg (n))^3 \implies \lg (g(n)) = \lg (n) + 3\lg (\lg (n))$ is logarithmic with respect to input because $\lg (n)$ dominates the expression.
\indent Since $\lg (f(n))$ and $\lg (g(n))$ are both polylogarithmic, but the power of $\lg (g(n))$ is greater than that of $\lg (f(n))$ $\implies \boxed{f(n) = O(g(n))}$

\vspace{3mm}

% Problem 1vii
\noindent vii. $f(n) = e^{\cos n} \textrm{ , } g(n) = \lg n$

\vspace{3mm}

\indent This problem is a bit confusing because you need to consider the range of the $\cos n$ function. Since the $\cos$ function has a range of $-1 \le \cos n \le 1$, that implies that the range of $e^{\cos n}$ is $\frac{1}{e} \le e^{\cos n} \le e$ and that f(n) is actually constant time with respect to the input.
\indent $g(n) = \lg n \implies g(n)$ is logarithmic
\indent Since logarithmic dominates constant time $\implies \boxed{f  = O(g)}$

\vspace{3mm}

% Problem 1viii
\noindent viii. $f(n) =  \lg n^2 \textrm{ , } g(n) = (\lg n)^2$

\vspace{3mm}

\indent $f(n) = \lg (n^2) = 2\lg (n)$ is logarithmic with respect to input.
\indent $g(n) = (\lg (n))^2$ is polylogarithmic with respect to input.
\indent Since g(n) is polylogarithmic and the power is greater than 1, g(n) dominates f(n) $\implies \boxed{f(n) = O(g(n))}$

\vspace{3mm}

% Problem 1ix
\noindent ix. $f(n) = \sqrt{4n^2 - 12n + 9} \textrm{ , } g(n) = n^{\frac{3}{2}}$

\vspace{3mm}

\indent $f(n) = \sqrt{4n^2 - 12n + 9} = \sqrt{(2n-3)^2} = 2n-3 \implies f(n)$ is linear
\indent $g(n) = n^{\frac{3}{2}} \implies g(n)$ is more than linear, but less than quadratic.
\indent Since f(n) and g(n) are both polynomial, but g(n) has a higher power than f(n) $\implies \boxed{f  = O(g)}$

\vspace{3mm}

% Problem 1x
\noindent x. $f(n) = \sum\limits_{k=1}^n k \textrm{ , } g(n) = (n+2)^2$

\vspace{3mm}

\indent $f(n) = \sum\limits_{k=1}^n k = \frac{n(n+1)}{2} = \frac{1}{2}n^2 + \frac{1}{2}n \implies f(n)$ is quadratic
\indent $g(n) = (n+2)^2 = n^2 + 4n + 4 \implies g(n)$ is quadratic
\indent Since f(n) is quadratic and g(n) is also quadratic $\implies \boxed{f  = \Theta(g)}$
